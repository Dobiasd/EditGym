import Data.List
import Data.Maybe
import Control.Arrow

main :: IO ()
main = do
    _ <- getLine
    interact $
        lines >>>
        filter (not . null) >>>
        map (read >>> posFstEven >>> show ) >>>
        unlines

posFstEven :: Int -> Int
posFstEven n
    | n < 3 = -1
    | otherwise = [2, 3, 2, 4] !! ((n -3) `mod` 4)
